    function escapeHtml(text) {

      return String(text)

        .split('&').join('&amp;')

        .split('<').join('&lt;')

        .split('>').join('&gt;')

        .split(QUOTE).join('&quot;');

    }


    function normalizeNewlines(text) {

      return String(text || '').split(CR + LF).join(LF).split(CR).join(LF);

    }


    function stripChordSymbols(text) {

      var input = String(text || '');

      var output = '';

      var insideQuote = false;

      for (var i = 0; i < input.length; i += 1) {

        var ch = input[i];

        if (ch === QUOTE) {

          insideQuote = !insideQuote;

        } else if (!insideQuote) {

          output += ch;

        }

      }

      return output;

    }


    function hasChordSymbols(text) {

      return String(text || '').indexOf(QUOTE) !== -1;

    }


    function isHeaderLine(line) {

      if (!line || line.length < 2) return false;

      var code = line.charCodeAt(0);

      return code >= 65 && code <= 90 && line[1] === ':';

    }


    function removeGraceNotes(text) {

      var output = '';

      var inside = false;

      for (var i = 0; i < text.length; i += 1) {

        if (text[i] === '{') {

          inside = true;

        } else if (text[i] === '}') {

          inside = false;

        } else if (!inside) {

          output += text[i];

        }

      }

      return output;

    }


    function melodyBodyText(rawAbc) {

      var lines = normalizeNewlines(stripChordSymbols(rawAbc)).split(LF);

      var body = [];

      for (var i = 0; i < lines.length; i += 1) {

        var line = lines[i].trim();

        if (!isHeaderLine(line)) body.push(line);

      }

      return removeGraceNotes(body.join(' '));

    }


    function cleanTokenText(text) {

      var out = '';

      for (var i = 0; i < text.length; i += 1) {

        var ch = text[i];

        if (ch === '|' || ch === '[' || ch === ']' || ch === '(' || ch === ')') out += ' ';

        else out += ch;

      }

      return out;

    }


    function tokenizeMelody(rawAbc) {

      var raw = cleanTokenText(melodyBodyText(rawAbc)).split(' ');

      var tokens = [];

      for (var i = 0; i < raw.length; i += 1) {

        var token = raw[i].trim();

        if (token) tokens.push(token);

      }

      return tokens;

    }


    function melodyBodyTextWithGrace(rawAbc) {

      var lines = normalizeNewlines(stripChordSymbols(rawAbc)).split(LF);

      var body = [];

      for (var i = 0; i < lines.length; i += 1) {

        var line = lines[i].trim();

        if (!isHeaderLine(line)) body.push(line);

      }

      return body.join(' ');

    }


    function tokenizeMelodyForPlayback(rawAbc) {

      var text = melodyBodyTextWithGrace(rawAbc);

      var tokens = [];

      var current = '';

      var i;


      function flushCurrent() {

        var token = current.trim();

        if (token) tokens.push(token);

        current = '';

      }


      for (i = 0; i < text.length; i += 1) {

        var ch = text[i];

        if (ch === '{') {

          flushCurrent();

          var grace = '';

          i += 1;

          while (i < text.length && text[i] !== '}') {

            grace += text[i];

            i += 1;

          }

          grace = grace.trim();

          if (grace) tokens.push('GRACE:' + grace);

        } else if (ch === '|' || ch === '[' || ch === ']' || ch === '(' || ch === ')' || ch === ' ') {

          flushCurrent();

        } else {

          current += ch;

        }

      }

      flushCurrent();

      return tokens;

    }


    function parseGraceToken(token) {

      var text = String(token || '');

      if (text.indexOf('GRACE:') !== 0) return null;

      return text.slice(6).trim();

    }


    function parseAbcPitch(token) {

      var text = String(token || '');

      var pos = 0;

      var accidental = 0;

      while (text[pos] === '^' || text[pos] === '_' || text[pos] === '=') {

        if (text[pos] === '^') accidental += 1;

        if (text[pos] === '_') accidental -= 1;

        pos += 1;

      }

      var note = text[pos];

      var base = { C: 60, D: 62, E: 64, F: 65, G: 67, A: 69, B: 71, c: 72, d: 74, e: 76, f: 77, g: 79, a: 81, b: 83 };

      if (base[note] === undefined) return null;

      var midi = base[note] + accidental;

      pos += 1;

      while (pos < text.length) {

        if (text[pos] === ',') midi -= 12;

        if (text[pos] === APOST) midi += 12;

        pos += 1;

      }

      return midi;

    }


    function parseAbcStaffStep(token) {

      var text = String(token || '');

      var pos = 0;

      while (text[pos] === '^' || text[pos] === '_' || text[pos] === '=') pos += 1;

      var note = text[pos];

      var scale = { C: 0, D: 1, E: 2, F: 3, G: 4, A: 5, B: 6 };

      if (!note || scale[note.toUpperCase()] === undefined) return null;

      var octave = note >= 'a' && note <= 'g' ? 5 : 4;

      pos += 1;

      while (pos < text.length) {

        if (text[pos] === ',') octave -= 1;

        if (text[pos] === APOST) octave += 1;

        pos += 1;

      }

      return octave * 7 + scale[note.toUpperCase()];

    }


    function staffYForStep(step, staffTop) {

      var e4 = 4 * 7 + 2;

      var distanceFromE4 = step - e4;

      var bottomLineY = staffTop + 40;

      return bottomLineY - distanceFromE4 * 5;

    }


    function abcNoteName(token) {

      var text = String(token || '');

      var pos = 0;

      var accidentalText = '';

      while (text[pos] === '^' || text[pos] === '_' || text[pos] === '=') {

        if (text[pos] === '^') accidentalText = '#';

        if (text[pos] === '_') accidentalText = 'b';

        pos += 1;

      }

      return accidentalText + String(text[pos] || '').toUpperCase();

    }


    function parseDurationBeats(token) {

      var text = String(token || '');

      var numberText = '';

      var slashIndex = text.indexOf('/');

      for (var i = 0; i < text.length; i += 1) {

        var ch = text[i];

        if (ch >= '0' && ch <= '9') numberText += ch;

        if (ch === '/') break;

      }

      var whole = numberText ? Number(numberText) : 1;

      if (slashIndex !== -1) return whole * 0.25;

      return whole * 0.5;

    }


    function getMelodyEventsFromAbc(rawAbc, includeGrace) {

      var tokens = includeGrace ? tokenizeMelodyForPlayback(rawAbc) : tokenizeMelody(rawAbc);

      var events = [];

      var beat = 0;

      var pendingGrace = [];

      var graceDuration = 0.50;

      var minMainAfterGrace = 1 / 3;


      for (var i = 0; i < tokens.length; i += 1) {

        var token = tokens[i];

        var graceToken = parseGraceToken(token);

        if (graceToken) {

          pendingGrace.push(graceToken);

          continue;

        }


        var duration = parseDurationBeats(token);

        if (token[0] === 'z') {

          events.push({ type: 'rest', beat: beat, duration: duration, label: 'z' });

          beat += duration;

          pendingGrace = [];

          continue;

        }


        var midi = parseAbcPitch(token);

        if (midi !== null) {

          var mainBeat = beat;

          var mainDuration = duration;


          if (includeGrace && pendingGrace.length) {

            var availableGrace = Math.max(0.08, duration - minMainAfterGrace);

            var actualGraceDuration = Math.min(graceDuration, availableGrace);

            var graceStart = beat;


            for (var g = 0; g < pendingGrace.length; g += 1) {

              var graceMidi = parseAbcPitch(pendingGrace[g]);

              if (graceMidi !== null) {

                events.push({

                  type: 'grace',

                  midi: graceMidi,

                  staffStep: parseAbcStaffStep(pendingGrace[g]),

                  beat: graceStart + g * actualGraceDuration,

                  duration: actualGraceDuration,

                  label: abcNoteName(pendingGrace[g])

                });

              }

            }


            mainBeat = beat + actualGraceDuration * pendingGrace.length;

            mainDuration = Math.max(minMainAfterGrace, duration - actualGraceDuration * pendingGrace.length);

          }


          events.push({ type: 'note', midi: midi, staffStep: parseAbcStaffStep(token), beat: mainBeat, duration: mainDuration, label: abcNoteName(token) });

          beat += duration;

          pendingGrace = [];

        }

      }

      return events;

    }


    function totalBeatsFromEvents(events) {

      if (!events.length) return 0;

      var last = events[events.length - 1];

      return last.beat + last.duration;

    }


    function isAlignedToFourBeats(events) {

      var total = totalBeatsFromEvents(events);

      return Math.abs(total - Math.round(total / 4) * 4) < 0.0001;

    }

      if (y < top - 4) {

          lineY -= 10;

        }

      }

    }


    function drawRest(parts, x, y, duration) {

      if (duration >= 1) {

        parts.push('<rect x="' + (x - 7) + '" y="' + (y - 4) + '" width="14" height="5" fill="#27231f"/>');

      } else {

        parts.push('<text x="' + (x - 7) + '" y="' + (y + 8) + '" font-size="24" font-family="serif" fill="#27231f">𝄾</text>');

      }

    }


    function noteHeadAttrs(duration) {

      if (duration >= 1) return { fill: '#fff', stroke: '#27231f', strokeWidth: 1.5, stem: false, flag: false };

      return { fill: '#27231f', stroke: '#27231f', strokeWidth: 1, stem: true, flag: duration <= 0.5 };

    }


    function renderScore(updateMessage) {

      try {

        var events = getMelodyEventsFromAbc(abcInput.value);

        if (!events.length) throw new Error('メロディー音が見つかりませんでした。');


        var measuresPerSystem = 4;

        var beatsPerMeasure = 4;

        var systemCount = Math.max(1, Math.ceil(totalBeatsFromEvents(events) / (measuresPerSystem * beatsPerMeasure)));

        var left = 78;

        var right = 18;

        var systemWidth = 920;

        var measureWidth = (systemWidth - left - right) / measuresPerSystem;

        var beatWidth = measureWidth / beatsPerMeasure;

        var measureInnerLeft = 22;

        var measureInnerRight = 6;

        var drawableMeasureWidth = measureWidth - measureInnerLeft - measureInnerRight;

        var drawableBeatWidth = drawableMeasureWidth / beatsPerMeasure;

        var systemHeight = 132;

        var titleHeight = 42;

        var width = systemWidth;

        var height = titleHeight + systemCount * systemHeight + 28;

        var parts = [];


        parts.push('<svg xmlns="http://www.w3.org/2000/svg" width="' + width + '" height="' + height + '" viewBox="0 0 ' + width + ' ' + height + '">');

        parts.push('<rect x="0" y="0" width="' + width + '" height="' + height + '" fill="#fff"/>');

        parts.push('<text x="24" y="28" font-size="18" font-family="serif" fill="#27231f">C Major Modern Jazz Etude</text>');


        for (var s = 0; s < systemCount; s += 1) {

          var staffTop = titleHeight + 38 + s * systemHeight;

          var systemEndX = width - right;

          parts.push('<text x="28" y="' + (staffTop + 42) + '" font-size="44" font-family="serif" fill="#27231f">𝄞</text>');

          for (var line = 0; line < 5; line += 1) {

            var yLine = staffTop + line * 10;

            parts.push('<line x1="' + left + '" y1="' + yLine + '" x2="' + systemEndX + '" y2="' + yLine + '" stroke="#27231f" stroke-width="1"/>');

          }

          for (var m = 0; m <= measuresPerSystem; m += 1) {

            var barX = left + m * beatsPerMeasure * beatWidth;

            parts.push('<line x1="' + barX + '" y1="' + staffTop + '" x2="' + barX + '" y2="' + (staffTop + 40) + '" stroke="#27231f" stroke-width="1"/>');

          }

        }


        for (var j = 0; j < events.length; j += 1) {

          var ev = events[j];

          var systemIndex = Math.floor(ev.beat / (measuresPerSystem * beatsPerMeasure));

          var localBeat = ev.beat - systemIndex * measuresPerSystem * beatsPerMeasure;

          var staffTopForEvent = titleHeight + 38 + systemIndex * systemHeight;

          var measureIndex = Math.floor(localBeat / beatsPerMeasure);

          var beatInMeasure = localBeat - measureIndex * beatsPerMeasure;

          var x = left + measureIndex * measureWidth + measureInnerLeft + beatInMeasure * drawableBeatWidth;

          if (beatInMeasure >= 3) x -= 1;


          if (ev.type === 'rest') {

            drawRest(parts, x, staffTopForEvent + 20, ev.duration);

            continue;

          }


          var y = staffYForStep(ev.staffStep, staffTopForEvent);

          var attrs = noteHeadAttrs(ev.duration);

          drawLedgerLines(parts, x, y, staffTopForEvent);

          parts.push('<ellipse cx="' + x + '" cy="' + y + '" rx="5.8" ry="4.1" fill="' + attrs.fill + '" stroke="' + attrs.stroke + '" stroke-width="' + attrs.strokeWidth + '" transform="rotate(-18 ' + x + ' ' + y + ')"/>');


          if (attrs.stem) {

            var stemTop = y - 32;

            parts.push('<line x1="' + (x + 5.5) + '" y1="' + y + '" x2="' + (x + 5.5) + '" y2="' + stemTop + '" stroke="#27231f" stroke-width="1.4"/>');

            if (attrs.flag) {

              parts.push('<path d="M ' + (x + 5.5) + ' ' + stemTop + ' C ' + (x + 20) + ' ' + (stemTop + 5) + ', ' + (x + 18) + ' ' + (stemTop + 17) + ', ' + (x + 7) + ' ' + (stemTop + 21) + '" fill="none" stroke="#27231f" stroke-width="1.25"/>');

            }

          }


          if (ev.label.indexOf('#') === 0) parts.push('<text x="' + (x - 19) + '" y="' + (y + 4) + '" font-size="16" fill="#27231f">#</text>');

          if (ev.label.indexOf('b') === 0) parts.push('<text x="' + (x - 17) + '" y="' + (y + 4) + '" font-size="16" fill="#27231f">b</text>');

        }


        parts.push('</svg>');

        scoreEl.innerHTML = parts.join('');

        visualObj = { ok: true, eventCount: events.length };

        if (updateMessage !== false) setStatus('五線譜を表示しました。', false);

      } catch (error) {

        console.error(error);

        visualObj = null;

        scoreEl.innerHTML = '<div style="padding:16px;color:#b45309;line-height:1.7;">五線譜に変換できませんでした。<br><small>' + escapeHtml(error.message || String(error)) + '</small></div>';

        setStatus('五線譜に変換できませんでした。', true);

      }

    }


    function parseChordText(text) {

      return normalizeNewlines(text)

        .split(LF)

        .join('|')

        .split('|')

        .map(function(item) { return item.trim(); })

        .filter(function(item) { return item.length > 0; });

    }


    function chordToMidiRoot(chordName) {

      var name = String(chordName || '').trim();

      var roots = { C: 60, D: 62, E: 64, F: 65, G: 67, A: 69, B: 71 };

      var root = roots[name[0]];

      if (root === undefined) return 60;

      if (name[1] === '#') root += 1;

      if (name[1] === 'b' || name[1] === '♭') root -= 1;

      return root;

    }


    function chordVoicing(chordName) {

      var name = String(chordName || '').toLowerCase();

      if (name.indexOf('dm7b5') !== -1) return [62, 65, 68, 72, 75];

      if (name.indexOf('g7alt') !== -1) return [65, 68, 70, 75, 80];

      if (name.indexOf('cmmaj7') !== -1 || name.indexOf('cmmaj') !== -1) return [63, 71, 74, 79, 86];

      if (name.indexOf('cm7') !== -1) return [63, 70, 74, 79, 82];

      if (name.indexOf('am7b5') !== -1) return [60, 67, 70, 75, 79];

      if (name.indexOf('dm7') !== -1) return [62, 65, 69, 72, 76];

      if (name.indexOf('cmaj7') !== -1 && name.indexOf('#11') !== -1) return [64, 71, 74, 78, 83];

      if (name.indexOf('cmaj7') !== -1) return [64, 71, 74, 79, 83];

      if (name.indexOf('am7') !== -1) return [64, 67, 71, 74, 76];

      var root = chordToMidiRoot(chordName);

      return [root + 4, root + 10, root + 14];

    }


    function getChordEventsFromChordInput() {

      var names = parseChordText(chordInput.value);

      return names.map(function(name, index) { return { name: name, beat: index * 4 }; });

    }


    function ensureAudioContext() {

      var AudioCtor = window.AudioContext || window.webkitAudioContext;

      if (!AudioCtor) throw new Error('Web Audio API is not available in this browser.');

      if (!audioContext || audioContext.state === 'closed') {

        audioContext = new AudioCtor();

        setupOutputChain(audioContext);

      }

      return audioContext;

    }


    function setupOutputChain(ctx) {

      masterGain = ctx.createGain();

      masterGain.gain.value = 1.0;

      reverbDelay = ctx.createDelay(1.0);

      reverbDelay.delayTime.value = 0.16;

      var feedback = ctx.createGain();

      feedback.gain.value = 0.26;

      var tone = ctx.createBiquadFilter();

      tone.type = 'lowpass';

      tone.frequency.value = 2400;

      reverbDelay.connect(feedback);

      feedback.connect(tone);

      tone.connect(reverbDelay);

      reverbDelay.connect(ctx.destination);

      masterGain.connect(ctx.destination);

    }


    function midiToFreq(midi) {

      return 440 * Math.pow(2, (midi - 69) / 12);

    }


    function makeDriveCurve(amount) {

      var samples = 256;

      var curve = new Float32Array(samples);

      var drive = Math.max(0, Math.min(1, amount || 0));

      var k = 30 * drive;

      for (var i = 0; i < samples; i += 1) {

        var x = i * 2 / samples - 1;

        curve[i] = (1 + k) * x / (1 + k * Math.abs(x));

      }

      return curve;

    }


    function scheduleTone(ctx, midi, start, duration, options) {

      var opts = Object.assign({ type: 'sine', gain: 0.1, filter: null, attack: 0.02, release: 0.05, sustain: 0.8, overlap: 0, slideFrom: null, slideTime: 0.055, reverbSend: 0.04, vibratoDepth: 0, vibratoRate: 5.2, drive: 0 }, options || {});

      var osc = ctx.createOscillator();

      var gain = ctx.createGain();

      var filter = opts.filter ? ctx.createBiquadFilter() : null;

      var shaper = opts.drive ? ctx.createWaveShaper() : null;

      var sendGain = ctx.createGain();

      var safeDuration = Math.max(0.08, duration + opts.overlap);

      var volume = Number(volumeInput.value || 0.75);

      var targetGain = opts.gain * volume;

      var attack = Math.min(opts.attack, safeDuration * 0.35);

      var endTime = start + safeDuration;

      var releaseTime = Math.min(opts.release, safeDuration * 0.55);

      var releaseStart = Math.max(start + attack + 0.02, endTime - releaseTime);

      var targetFreq = midiToFreq(midi);


      osc.type = opts.type;

      if (opts.vibratoDepth > 0) {

        var lfo = ctx.createOscillator();

        var lfoGain = ctx.createGain();

        lfo.type = 'sine';

        lfo.frequency.setValueAtTime(opts.vibratoRate, start);

        lfoGain.gain.setValueAtTime(0, start);

        lfoGain.gain.linearRampToValueAtTime(opts.vibratoDepth, start + Math.min(0.18, safeDuration * 0.45));

        lfoGain.gain.linearRampToValueAtTime(opts.vibratoDepth * 0.75, endTime);

        lfo.connect(lfoGain);

        lfoGain.connect(osc.detune);

        lfo.start(start);

        lfo.stop(endTime + 0.08);

        activeNodes.push(lfo);

      }

      if (opts.slideFrom !== null) {

        osc.frequency.setValueAtTime(midiToFreq(opts.slideFrom), start);

        osc.frequency.exponentialRampToValueAtTime(targetFreq, start + Math.min(opts.slideTime, safeDuration * 0.72));

      } else {

        osc.frequency.setValueAtTime(targetFreq, start);

      }


      gain.gain.setValueAtTime(0.0001, start);

      gain.gain.linearRampToValueAtTime(targetGain, start + attack);

      gain.gain.setValueAtTime(targetGain * opts.sustain, releaseStart);

      gain.gain.linearRampToValueAtTime(0.0001, endTime);


      if (shaper) {

        shaper.curve = makeDriveCurve(opts.drive);

        shaper.oversample = '2x';

      }


      if (filter) {

        filter.type = opts.filter.type;

        filter.frequency.setValueAtTime(opts.filter.frequency, start);

        filter.Q.setValueAtTime(opts.filter.q || 1, start);

        if (shaper) {

          osc.connect(shaper);

          shaper.connect(filter);

        } else {

          osc.connect(filter);

        }

        filter.connect(gain);

      } else {

        if (shaper) {

          osc.connect(shaper);

          shaper.connect(gain);

        } else {

          osc.connect(gain);

        }

      }


      sendGain.gain.value = opts.reverbSend;

      gain.connect(masterGain || ctx.destination);

      if (reverbDelay) {

        gain.connect(sendGain);

        sendGain.connect(reverbDelay);

      }

      osc.start(start);

      osc.stop(endTime + 0.08);

      activeNodes.push(osc);

    }


    function scheduleSlurTone(ctx, fromMidi, toMidi, start, graceDuration, mainDuration, options) {

      var opts = Object.assign({ type: 'sawtooth', gain: 0.13, filter: { type: 'bandpass', frequency: 760, q: 1.35 },

              vibratoDepth: 4.5,

              vibratoRate: 5.4,

              drive: 0.16, attack: 0.018, release: 0.16, reverbSend: 0.06, vibratoDepth: 0, vibratoRate: 5.2, drive: 0 }, options || {});

      var osc = ctx.createOscillator();

      var gain = ctx.createGain();

      var filter = opts.filter ? ctx.createBiquadFilter() : null;

      var shaper = opts.drive ? ctx.createWaveShaper() : null;

      var sendGain = ctx.createGain();

      var volume = Number(volumeInput.value || 0.75);

      var targetGain = opts.gain * volume;

      var totalDuration = Math.max(0.12, graceDuration + mainDuration);

      var endTime = start + totalDuration;

      var graceHoldEnd = start + Math.max(0.025, graceDuration * 0.56);

      var slideEnd = start + Math.max(0.05, graceDuration * 0.96);

      var releaseStart = Math.max(start + 0.04, endTime - opts.release);


      osc.type = opts.type;

      if (opts.vibratoDepth > 0) {

        var lfo = ctx.createOscillator();

        var lfoGain = ctx.createGain();

        lfo.type = 'sine';

        lfo.frequency.setValueAtTime(opts.vibratoRate, start);

        lfoGain.gain.setValueAtTime(0, start);

        lfoGain.gain.linearRampToValueAtTime(opts.vibratoDepth, start + Math.min(0.2, totalDuration * 0.45));

        lfoGain.gain.linearRampToValueAtTime(opts.vibratoDepth * 0.75, endTime);

        lfo.connect(lfoGain);

        lfoGain.connect(osc.detune);

        lfo.start(start);

        lfo.stop(endTime + 0.08);

        activeNodes.push(lfo);

      }

      osc.frequency.setValueAtTime(midiToFreq(fromMidi), start);

      osc.frequency.setValueAtTime(midiToFreq(fromMidi), graceHoldEnd);

      osc.frequency.exponentialRampToValueAtTime(midiToFreq(toMidi), slideEnd);

      osc.frequency.setValueAtTime(midiToFreq(toMidi), slideEnd);


      gain.gain.setValueAtTime(0.0001, start);

      gain.gain.linearRampToValueAtTime(targetGain * 0.94, start + opts.attack);

      gain.gain.setValueAtTime(targetGain * 0.94, graceHoldEnd);

      gain.gain.linearRampToValueAtTime(targetGain, slideEnd + 0.01);

      gain.gain.setValueAtTime(targetGain * 0.88, releaseStart);

      gain.gain.linearRampToValueAtTime(0.0001, endTime);


      if (shaper) {

        shaper.curve = makeDriveCurve(opts.drive);

        shaper.oversample = '2x';

      }


      if (filter) {

        filter.type = opts.filter.type;

        filter.frequency.setValueAtTime(opts.filter.frequency, start);

        filter.Q.setValueAtTime(opts.filter.q || 1, start);

        if (shaper) {

          osc.connect(shaper);

          shaper.connect(filter);

        } else {

          osc.connect(filter);

        }

        filter.connect(gain);

      } else {

        if (shaper) {

          osc.connect(shaper);

          shaper.connect(gain);

        } else {

          osc.connect(gain);

        }

      }


      sendGain.gain.value = opts.reverbSend;

      gain.connect(masterGain || ctx.destination);

      if (reverbDelay) {

        gain.connect(sendGain);

        sendGain.connect(reverbDelay);

      }

      osc.start(start);

      osc.stop(endTime + 0.08);

      activeNodes.push(osc);

    }


    function stopAudio(updateStatus) {

      timers.forEach(function(timerId) { window.clearTimeout(timerId); });

      timers = [];

      activeNodes.forEach(function(node) { try { node.stop(); } catch (error) {} });

      activeNodes = [];


      // スマホではAudioContextを閉じると、次のタップで再開できない場合があるため閉じない。

      if (updateStatus) setStatus('停止しました。', false);

    }


    function playDirectBeep(ctx, frequency, duration, gainValue) {

      var now = ctx.currentTime;

      var osc = ctx.createOscillator();

      var gain = ctx.createGain();

      osc.type = 'sine';

      osc.frequency.setValueAtTime(frequency || 880, now);

      gain.gain.setValueAtTime(0.0001, now);

      gain.gain.linearRampToValueAtTime(gainValue || 0.7, now + 0.02);

      gain.gain.setValueAtTime(gainValue || 0.7, now + Math.max(0.04, duration - 0.08));

      gain.gain.linearRampToValueAtTime(0.0001, now + duration);

      osc.connect(gain);

      gain.connect(ctx.destination);

      osc.start(now);

      osc.stop(now + duration + 0.03);

      activeNodes.push(osc);

    }

    function playAudio() {

      try {

        stopAudio(false);

        var allEvents = getMelodyEventsFromAbc(abcInput.value, true);

        var melodyEvents = allEvents.filter(function(event) { return event.type === 'note' || event.type === 'grace'; });

        var chordEvents = getChordEventsFromChordInput();


        if (!melodyEvents.length) {

          setStatus('再生できるメロディー音が見つかりませんでした。', true);

          return;

        }


        unlockAudio().then(function(ctx) {

          var tempo = Number(tempoInput.value || 112);

          var secondsPerBeat = 60 / tempo;

          var now = ctx.currentTime + 0.08;


          for (var melodyIndex = 0; melodyIndex < melodyEvents.length; melodyIndex += 1) {

            var event = melodyEvents[melodyIndex];

            var nextMelody = melodyEvents[melodyIndex + 1];

            var previous = melodyEvents[melodyIndex - 1];

            var isGrace = event.type === 'grace';

            var followsGrace = previous && previous.type === 'grace';


            if (isGrace && nextMelody && nextMelody.type === 'note') {

              scheduleSlurTone(

                ctx,

                event.midi,

                nextMelody.midi,

                now + event.beat * secondsPerBeat,

                event.duration * secondsPerBeat,

                nextMelody.duration * secondsPerBeat * 0.96,

                {

                  type: 'triangle',

                  gain: 0.16,

                  attack: 0.006,

                  release: 0.36,

                  reverbSend: 0.045,

                  filter: { type: 'lowpass', frequency: 2600, q: 0.35 },

                  vibratoDepth: 0,

                  vibratoRate: 5.4,

                  drive: 0

                }

              );

              melodyIndex += 1;

              continue;

            }


            var isLegatoNote = previous && previous.type === 'note' && Math.abs(previous.midi - event.midi) <= 7;

            var slideFrom = isLegatoNote ? previous.midi : null;

            scheduleTone(ctx, event.midi, now + event.beat * secondsPerBeat, event.duration * secondsPerBeat * 0.98, {

              type: 'triangle',

              gain: followsGrace ? 0.15 : 0.14,

              attack: followsGrace ? 0.002 : 0.008,

              release: isLegatoNote ? 0.42 : 0.34,

              sustain: 0.54,

              overlap: isLegatoNote ? secondsPerBeat * 0.04 : 0,

              slideFrom: null,
